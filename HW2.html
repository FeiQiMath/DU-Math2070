    Section 3.2 Homework 2

    1. Find the solution of each of the following IVP.
      1. \(t^2 y' + 2ty = \cos t, y(\pi)= 0, t\gt 0. \)
        Solution.
        \begin{equation*} y' + \frac 2 t y = \frac{\cos t}{t^2}. \end{equation*}
        \begin{equation*} p(t)=\frac 2 t, g(t) = \frac{\cos t}{t^2}. \end{equation*}
        \begin{equation*} \mu(t) = e^{\int \frac 2 t dt} = e^{2\ln t} = t^2 \end{equation*}
        \(\textbf{Warning! Do not abuse algebra! }e^{2\ln t} \textbf{ is }t^2, \textbf{ not }2t.\)
        \begin{align*} y(t) \amp = \frac{\int \mu(t) g(t) dt}{\mu(t)}=\displaystyle{\frac{\int t^2 \cdot \frac{\cos t}{t^2} dt}{t^2}}\\ \amp= \dfrac{\int \cos t dt}{t^2} = \dfrac{\sin t + C}{t^2} \end{align*}
        \begin{equation*} y(\pi)=0\Rightarrow \dfrac{\sin\pi + C}{\pi^2}=0\Rightarrow C = 0. \end{equation*}
        \begin{equation*} y(t) = \frac{\sin t}{t^2}. \end{equation*}
      2. \(ty'+3y = \cos t, y(\pi) = 0, t \gt 0. \)
        Solution.
        \begin{equation*} y' + \frac 3 t y = \frac{\cos t}{t}. \end{equation*}
        \begin{equation*} p(t) = \frac 3 t, g(t) = \frac{\cos t}{t}. \end{equation*}
        \begin{equation*} \mu(t) = e^{\int \frac 3 t dt} = e^{3\ln t} = t^3. \end{equation*}
        \begin{equation*} y(t) = \frac{\int \mu(t)g(t)dt}{\mu(t)} = \frac{\int t^3 \frac{\cos t}{t}}{t^3}= \frac{\int t^2 \cos t dt}{t^3}. \end{equation*}
        \begin{align*} \int t^2 \cos t dt \amp = t^2 \sin t - \int 2t \sin t dt\\ \amp = t^2 \sin t - 2\left(t\cdot (-\cos t) - \int 1 \cdot (-\cos t)dt\right)\\ \amp = t^2 \sin t + 2t \cos t - 2\sin t + C \end{align*}
        \begin{equation*} y(t) = \frac {\sin t} t + \frac{2\cos t}{t^2} - \frac{2\sin t}{t^3}+ \frac{C}{t^3}. \end{equation*}
        \begin{equation*} y(\pi) = 0\Rightarrow 0 + \frac{-2}{\pi^2} - 0 + \frac{C}{\pi^3} = 0 \Rightarrow C = 2\pi. \end{equation*}
        \begin{equation*} y(t) = \frac {\sin t} t + \frac{2\cos t}{t^2} - \frac{2\sin t}{t^3}- \frac{2\pi}{t^3}. \end{equation*}
      3. \(ty' + (1+t)y = e^{-t}\sin 2t, y(\pi) = 0, t\gt 0. \)
        Solution.
        \begin{equation*} y' + \frac {1+t} t y = \frac{e^{-t}\sin 2t}{t}. \end{equation*}
        \begin{equation*} p(t) = \frac {1+t} t, g(t) = \frac{e^{-t}\sin 2t}{t}. \end{equation*}
        \begin{equation*} \mu(t) = e^{\int p(t)dt} = e^{\int \frac {1+t} t dt} = e^{\int \left(\frac 1 t + 1\right) dt} = e^{\ln t + t} = te^t. \end{equation*}
        \(\textbf{Warning! Do not abuse algebra!} e^{A+B} = e^A\cdot e^B, \textbf{ not }e^A + e^B. \)
        \begin{equation*} y(t) = \frac{\int \mu(t)g(t)dt}{\mu(t)} = \frac{\int te^t \frac{e^{-t}\sin 2t}{t}}{te^t}= \frac{\int \sin 2t dt}{te^t} = \frac{-\cos 2t + C}{2te^t}. \end{equation*}
        \begin{align*} \int t^2 \cos t dt \amp = t^2 \sin t - \int 2t \sin t dt\\ \amp = t^2 \sin t - 2\left(t\cdot (-\cos t) - \int 1 \cdot (-\cos t)dt\right)\\ \amp = t^2 \sin t + 2t \cos t - 2\sin t + C \end{align*}
        \begin{equation*} y(t) = \frac {\sin t} t + \frac{2\cos t}{t^2} - \frac{2\sin t}{t^3}+ \frac{C}{t^3}. \end{equation*}
        \begin{equation*} y(\pi) = 0\Rightarrow 0 + \frac{-2}{\pi^2} - 0 + \frac{C}{\pi^3} = 0 \Rightarrow C = 2\pi. \end{equation*}
        \begin{equation*} y(t) = \frac {\sin t} t + \frac{2\cos t}{t^2} - \frac{2\sin t}{t^3}- \frac{2\pi}{t^3}. \end{equation*}
    2. Find the solution of each of the following IVP that is continuous everywhere.
      1. \begin{equation*} y' + y = g(t), y(0) = 1, g(t) = \begin{cases} t \amp \text{if } 0 \leq t \lt 1\\ 0 \amp \text{if }t \geq 1 \end{cases} \end{equation*}
      2. \begin{equation*} (1+t^2) y' + 2t y = g(t), y(0) = 0, g(t) = \begin{cases} t \amp \text{if } 0 \leq t \lt 1\\ -t \amp \text{if }t \geq 1 \end{cases} \end{equation*}
    3. Find a one-parameter family of solutions of the following ODEs.
      1. \(\displaystyle \frac{dy}{dx}=\frac{x-e^{-x}}{y+e^y}.\)
      2. \(\displaystyle xy' = \sqrt{9-y^2}.\)
      3. \(\displaystyle \frac{dy}{dx}=\frac{xy+2y-x-2}{xy-3y+x-3}.\)
    4. Find the \(\textit{explicit}\) solution of the following IVPs and the interval of existence.
      1. \(\displaystyle y' = (1-2x)y^2, y(0)=-1/6.\)
      2. \(\displaystyle y' = \frac{x(x^2+1)}{4y^3}, y(0)=-1/\sqrt{2}.\)
      3. \(\displaystyle (2y-2)y'= 3x^2+4x+2, y(1)=-2.\)
